Data Processing

study2012 <- read.csv("pisa2012.csv", na.strings=c(7,8,9))

sample_turkey <- study2012[study2012$ï..CNT == 'TUR',]
sample_turkey <- na.omit(sample_turkey)

# recode & derive new variables (scores)
PV1MATH = as.numeric(as.character(sample_turkey$PV1MATH))
PV2MATH = as.numeric(as.character(sample_turkey$PV2MATH))
PV3MATH = as.numeric(as.character(sample_turkey$PV3MATH))
PV4MATH = as.numeric(as.character(sample_turkey$PV4MATH))
PV5MATH = as.numeric(as.character(sample_turkey$PV5MATH))

MATH = (PV1MATH + PV2MATH + PV3MATH + PV4MATH + PV5MATH)/5
Interest = sample_turkey$ST29Q01 + sample_turkey$ST29Q03 + sample_turkey$ST29Q04+ sample_turkey$ST29Q06
Motivation = sample_turkey$ST29Q02 + sample_turkey$ST29Q05 + sample_turkey$ST29Q07 + sample_turkey$ST29Q08
Attitudes = sample_turkey$IC22Q01 + sample_turkey$IC22Q02 +  sample_turkey$IC22Q04 + sample_turkey$IC22Q06 + sample_turkey$IC22Q07 +  sample_turkey$IC22Q08
Self_efficacy = sample_turkey$ST37Q01 +sample_turkey$ST37Q02+ sample_turkey$ST37Q03 + sample_turkey$ST37Q04 + sample_turkey$ST37Q05+ sample_turkey$ST37Q06+ sample_turkey$ST37Q07+ sample_turkey$ST37Q08
Anxiety = sample_turkey$ST42Q01 + sample_turkey$ST42Q03 + sample_turkey$ST42Q05 + sample_turkey$ST42Q08 
Perception = sample_turkey$ST42Q02 + sample_turkey$ST42Q04 + sample_turkey$ST42Q06 + sample_turkey$ST42Q07+ sample_turkey$ST42Q09
Discipline = sample_turkey$ST46Q01 + sample_turkey$ST46Q02 + sample_turkey$ST46Q03+ sample_turkey$ST46Q04+ sample_turkey$ST46Q05+ sample_turkey$ST46Q06+ sample_turkey$ST46Q07+ sample_turkey$ST46Q08+ sample_turkey$ST46Q09

MATH = (MATH>mean(MATH))*1 #math literacy (or DV) 1 = Above average, 0 = Below average 

myList = list(interst = Interest, 
              motivation = Motivation, 
              attitudes = Attitudes, 
              self_efficacy = Self_efficacy,
              anxiety = Anxiety, 
              perception = Perception, 
              discipline = Discipline, 
              math = MATH)
data = as.data.frame(myList) # final data
